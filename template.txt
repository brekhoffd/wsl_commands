mkdir C:\wsl\backup\
mkdir C:\wsl\debian\
mkdir C:\wsl\ubuntu\
wsl --install -d Debian
wsl --export Debian C:\wsl\backup\debian.tar
wsl --unregister Debian
wsl --install -d Ubuntu
wsl --export Ubuntu C:\wsl\backup\ubuntu.tar
wsl --unregister Ubuntu
wsl --import debian C:\wsl\debian\ C:\wsl\backup\debian.tar --version 2
wsl --import ubuntu C:\wsl\ubuntu\ C:\wsl\backup\ubuntu.tar --version 2
wsl -d debian
wsl -d ubuntu
wsl --list --verbose
wsl --shutdown
